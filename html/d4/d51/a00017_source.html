26  static double parseVal(const std::string &value ){
27  std::size_t suffixPos = value.find_first_not_of("0.123456789");
28  if( suffixPos == std::string::npos ){
29  return std::stod( value );
30  }
31  std::string unitSuffix = value.substr(suffixPos, std::string::npos);
32  int mult;
33  // std::cerr<<(int)unitSuffix[0]<<" "<<unitSuffix<<std::endl;
34 
35  if (unitSuffix == "p"){
36  mult = -12;
37  }
38  else if (unitSuffix == "n"){
39  mult = -9;
40  }
41  else if (unitSuffix == "u" || unitSuffix == "Âµ"){
42  mult = -6;
43  }
44  else if (unitSuffix == "m"){
45  mult = -3;
46  }
47  else if (unitSuffix == "k"){
48  mult = 3;
49  }
50  else if (unitSuffix == "Meg"){
51  mult = 6;
52  }
53  else if (unitSuffix == "G"){
54  mult = 9;
55  }
56  else {
57  std::cerr << "Invalid Unit Suffix" << '\n';
58  assert(0);
59  }
60  std::string num = value.substr(0, suffixPos);
61  return std::stod(num) * pow(10, mult);
62  }
